# 1. Initialize the local repository
git init

# 2. Add the remote link (pointing to your existing GitHub repo)
git remote add origin https://github.com/mojoxanadu/skip-work.git

# 3. Rename the default branch to 'main' to match GitHub
git branch -M main

# 4. Stage every file we've created (Code, Logo, and Gradle)
git add .

# 5. Create your first commit
git commit -m "Initial Alpha: Title screen, Isometric logo, and Gradle build system"

# 6. Push to GitHub
git push -u origin main
